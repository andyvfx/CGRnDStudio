import pymel.core as pm

def format_(str):
    return "|{0: <50}".format(str)
def format_s():
    return "|"+"-"*51
node = "hairSystemShape1"
print len(pm.listAttr(node))
print format_s()*3
for each in sorted(pm.listAttr(node)):
    attr = ".".join([node, each])
    nice_name = pm.attributeName(attr, nice=1)
    short_name = pm.attributeName(attr, short=1)
    print format_(nice_name), format_(each), format_(short_name)
    print format_s()*3
